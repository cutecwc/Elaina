使用示例：

```makefile
cc = g++
co = -c -Wall
target = main.o mkt.o
file_cpp = ./cpp/


all: main

main:main.o mkt.o
	$(cc) $(target) -o main


main.o: main.cpp
	$(cc) $(co) main.cpp
mkt.o: $(file_cpp)mkt.cpp
	$(cc) $(co) $(file_cpp)mkt.cpp


clean:
	rm -f *.o main
```
